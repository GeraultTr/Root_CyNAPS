-> 3. When a lot of properties are to be declared

# New properties' creation in MTG
self.keywords = dict(	Nm=Nm,
                        AA=AA,
                        etc=etc
                        )

props = g.properties()
for name in self.keywords:
	props.setdefault(name, {})

# vertices storage for future calls in for loops
self.vertices = self.g.vertices(scale=g.max_scale())
for vid in self.vertices:
	for name, value in self.keywords.items():
		# Effectively creates the new property
		props[name][vid] = value

# Accessing properties once for future calls in for loops, execution time /5
self.states += """
			Nm
			AA
			length # already existing property
			radius # already existing property
			etc
			""".split()

for name in self.states:
	setattr(self, name, props[name])
